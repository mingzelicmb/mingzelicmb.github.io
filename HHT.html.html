100          function cmykToRgb(c, m, y, k) {
101              let r = 255 * (1 - c) * (1 - k);
102              let g = 255 * (1 - m) * (1 - k);
103              let b = 255 * (1 - y) * (1 - k);
104              return [r, g, b];
105          }
106  
107          function hsvToRgb(h, s, v) {
108              let r, g, b;
109              let i = Math.floor(h * 6);
110              let f = h * 6 - i;
111              let p = v * (1 - s);
112              let q = v * (1 - f * s);
113              let t = v * (1 - (1 - f) * s);
114              switch (i % 6) {
115                  case 0: r = v, g = t, b = p; break;
116                  case 1: r = q, g = v, b = p; break;
117                  case 2: r = p, g = v, b = t; break;
118                  case 3: r = p, g = q, b = v; break;
119                  case 4: r = t, g = p, b = v; break;
120                  case 5: r = v, g = p, b = q; break;
121              }
122              return [r * 255, g * 255, b * 255];
123          }
